`Time complexity : O(n)`

`Space complexity : O(n)`
```python
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        if head is None:
            return False
        slow, fast = head, head.next
        
        while fast and fast.next not in [None]:
            if slow == fast:
                return True
            slow = slow.next
            fast = fast.next.next
        
        return False
```
